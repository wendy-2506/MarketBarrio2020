CREATE TABLE Tipo_Empleado (
	idTipoEmp int NOT NULL, 
	descripcion varchar(255),
	PRIMARY KEY(idTipoEmp)
);

CREATE TABLE Empleado(
	idEmpleado int NOT NULL,
	apellidos varchar(255),
	nombres varchar(255),
	idEmpReg int,
	fechReg date,
	idEmpMod int,
	fechMod date,
	usr varchar(255),
	pw varchar (255),
	idTipoEmp int,
	estado int,
	PRIMARY KEY (idEmpleado),
	FOREIGN KEY (idTipoEmp) REFERENCES Tipo_Empleado (idTipoEmp),
	FOREIGN KEY (idEmpMod ) REFERENCES Empleado(idEmpleado),
	FOREIGN KEY (idEmpReg ) REFERENCES Empleado(idEmpleado)
);
CREATE TABLE Proveedor (
	idProveedor int NOT NULL,
	razon_Soc varchar(255),
	ruc varchar (255),
	nomb_repre varchar (255),
	direccion varchar (255),
	telefono varchar(255),
	PRIMARY KEY (idProveedor)
);
CREATE TABLE Tienda(
	idTienda int NOT NULL,
	descTienda varchar(255),
	direccion varchar(255),
	PRIMARY KEY (idTienda)
);

CREATE TABLE Categoria(
	idCategoria int NOT NULL,
	descrip varchar(255),
	PRIMARY KEY (idCategoria)
);

CREATE TABLE Marca(
	idMarca int NOT NULL,
	descMarc varchar(255),
	PRIMARY KEY (idMarca)
);


CREATE TABLE Tipo_Cliente(
	idTipoCli int NOT NULL,
	descripcion varchar(255),
	PRIMARY KEY (idTipoCli)
);


CREATE TABLE Cliente(
	idCliente int NOT NULL,
	idTipoCli int,
	idEmpReg int ,
	fechReg date,
	idEmpMod int ,
	fechMod date,
	correo_electr varchar(300),
	telefono varchar(255),
	direccion varchar(255),
	estado int,
	PRIMARY KEY (idCliente),
	FOREIGN KEY (idTipoCli ) REFERENCES Tipo_Cliente(idTipoCli),
	FOREIGN KEY (idEmpReg ) REFERENCES Empleado(idEmpleado),
	FOREIGN KEY (idEmpMod ) REFERENCES Empleado(idEmpleado)
);

CREATE TABLE Cliente_Persona(
	idCliente int,
	apell varchar(255),
	nomb varchar (255),
	dni float,
	FOREIGN KEY (idCliente) REFERENCES Cliente(idCliente)
);

CREATE TABLE Cliente_Empresa(
	idCliente int,
	razon_Social varchar (255),
	representante varchar (255),
	ruc varchar (255),
	FOREIGN KEY (idCliente) REFERENCES Cliente(idCliente)
);

CREATE TABLE SerieDV (
    idSerie int NOT NULL,
	idEmpReg int ,
    	fechReg date,
	idTienda int ,
	numInicio int,
	numFin int,
	fechCierre date,
   	estado int,
	PRIMARY KEY (idSerie),
	FOREIGN KEY (idEmpReg ) REFERENCES Empleado(idEmpleado),
	FOREIGN KEY (idTienda ) REFERENCES Tienda(idTienda)
);

CREATE TABLE Cab_Entrada (
    idEntrada int NOT NULL,
	idEmpReg int ,
    	fechReg date,
	idTienda int ,
	idProveedor int ,
	idEmpMod int ,
	fechMod date,
	estado int,
	PRIMARY KEY (idEntrada),
	FOREIGN KEY (idEmpReg ) REFERENCES Empleado(idEmpleado),
	FOREIGN KEY (idTienda ) REFERENCES Tienda(idTienda),
	FOREIGN KEY (idProveedor ) REFERENCES Proveedor(idProveedor),
	FOREIGN KEY (idEmpMod ) REFERENCES Empleado(idEmpleado)
);

CREATE TABLE Producto(
	idProducto int NOT NULL,
	descripcion varchar(255),
	idCategoria int ,
	idMarca int ,
	precioUnit float,
	unidMed varchar(255),
	estado int,
	PRIMARY KEY(idProducto),
	FOREIGN KEY (idCategoria ) REFERENCES Categoria(idCategoria),
	FOREIGN KEY (idMarca )REFERENCES Marca(idMarca)
);


CREATE TABLE Det_Entrada(
	idEntrada int ,
	idProducto int ,
	precio float,
	cantidad int,
	FOREIGN KEY (idEntrada) REFERENCES Cab_Entrada(idEntrada),
	FOREIGN KEY (idProducto) REFERENCES Producto(idProducto)
);


CREATE TABLE TipDocVenta(
	idTipo int NOT NULL,
	descrip varchar(255),
	PRIMARY KEY (idTipo)
);


CREATE TABLE Cab_DocVenta(
	idVenta int NOT NULL,
	idCliente int ,
	idEmpReg int ,
	fechVenta date,
	idEmpMod int ,
	fechMod date,
	idTipo int ,
	idTienda int ,
	estado int,
	PRIMARY KEY (idVenta),
	FOREIGN KEY(idEmpReg ) REFERENCES Empleado(idEmpleado),
	FOREIGN KEY(idCliente ) REFERENCES Cliente(idCliente),
	FOREIGN KEY (idEmpMod )REFERENCES Empleado(idEmpleado),
	FOREIGN KEY (idTipo )REFERENCES TipDocVenta(idTipo),
	FOREIGN KEY (idTienda )REFERENCES Tienda(idTienda)
);

CREATE TABLE Factura(
	idFactura int NOT NULL,
	idSerie int ,
	numFac varchar(50) ,
	idVenta int ,
	PRIMARY KEY (idFactura),
	FOREIGN KEY (idSerie ) REFERENCES SerieDV(idSerie),
	FOREIGN KEY (idVenta ) REFERENCES Cab_DocVenta(idVenta)
);

CREATE TABLE Boleta(
	idBoleta int NOT NULL,
	idSerie int ,
	numBol varchar(50) ,
	idVenta int ,
	PRIMARY KEY (idBoleta),
	FOREIGN KEY (idSerie ) REFERENCES SerieDV(idSerie),
	FOREIGN KEY (idVenta )REFERENCES Cab_DocVenta(idVenta)
);


CREATE TABLE Det_DocVenta(
	idVenta int ,
	idProducto int ,
	precio float,
	cantidad int,
	FOREIGN KEY (idVenta)REFERENCES Cab_DocVenta(idVenta),
	FOREIGN KEY (idProducto)REFERENCES Producto(idProducto)
);

CREATE TABLE ProdStock (
	idTienda int ,
	idProducto int ,
	stock int,
	FOREIGN KEY (idTienda) REFERENCES Tienda(idTienda),
	FOREIGN KEY (idProducto) REFERENCES Producto(idProducto)
);
